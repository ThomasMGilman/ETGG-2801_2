layout(location=0) in vec3 position;
layout(location=1) in vec2 texCoord;
out vec2 v_texCoord;

void main(){
    vec3 transform = vec3(position);
    transform *= worldMatrix;
    transform *= viewMatrix;
    transform *= projMatrix;


    gl_Position = vec4( transform.xyz, 1 );
    gl_PointSize = 1;
    v_texCoord = texCoord;
}
